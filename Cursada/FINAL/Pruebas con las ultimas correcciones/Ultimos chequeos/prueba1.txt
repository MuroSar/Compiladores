a,b:LONG.

a=1.
b=1.

LONG FUNCTION f1 {
	b=10.
	c,d:LONG.
	c=1.
	d=a*b+c+2*b+a*b+c.
	RETURN (d).
}

IF (f1() == 42) THEN
OUT ("Anda 1").
ELSE 
OUT("Error 1").
END_IF.

IF (f1() == 10) THEN
OUT ("Error 2").
ELSE 
OUT("Anda 2").
END_IF.

	a,b,c,d:LONG.
	a=30.
	b=10.
	c=1.
	d=a*b+c+2*b+a*b+c.
	RETURN (d).
}
